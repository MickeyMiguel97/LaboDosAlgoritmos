typedef struct nodo_s{
    int dato;
    struct nodo_s *siguiente;
}nodo_t;

typedef nodo_t *ptrNodo;
typedef nodo_t *ptrPila;

void push(ptrPila *pila, int x){
    ptrNodo nd;
    nd = (ptrNodo)malloc(sizeof(nodo_t));
    if (nd != NULL){
        nd -> dato = x;
        nd -> siguiente = *pila;
        *pila = nd;
    }
}
